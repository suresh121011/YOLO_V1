## Core Dataclasses

```python
# src/pipeline/__init__.py — Shared data contracts

from dataclasses import dataclass, field
from typing import Optional
from enum import Enum


class Severity(Enum):
    INFO = 0
    LOW = 1
    MEDIUM = 2
    HIGH = 3
    CRITICAL = 4


@dataclass
class BoundingBox:
    """Normalized bounding box in YOLO format (cx, cy, w, h)."""
    cx: float  # center x in [0, 1]
    cy: float  # center y in [0, 1]
    w: float   # width in [0, 1]
    h: float   # height in [0, 1]

    def area(self) -> float:
        return self.w * self.h

    def iou(self, other: "BoundingBox") -> float:
        """Compute intersection-over-union."""
        ...


@dataclass
class Detection:
    """Single object detection from YOLO."""
    class_id: int
    class_name: str
    confidence: float          # [0.0, 1.0]
    bbox: BoundingBox
    frame_id: int
    timestamp_ms: float


@dataclass
class SceneContext:
    """Output from SmolVLM2 scene analysis."""
    activity: str
    risks: list[dict]          # [{"risk_type": str, "severity": str, "description": str}]
    recommendations: list[str]
    raw_response: str
    inference_time_ms: float
    frame_id: int


@dataclass
class Alert:
    """Safety alert generated by Rule Engine."""
    rule_id: str
    severity: Severity
    message: str
    message_hi: Optional[str]  # Hindi translation (V2)
    triggering_detections: list[Detection]
    timestamp_ms: float
    cooldown_seconds: int
    frame_id: int


@dataclass
class MemoryEntry:
    """Temporal tracking entry for Event Memory."""
    class_id: int
    class_name: str
    last_seen_frame: int
    first_seen_frame: int
    detection_count: int
    avg_confidence: float
    last_bbox: BoundingBox


@dataclass
class PipelineMetrics:
    """Per-frame performance metrics."""
    frame_id: int
    capture_ms: float
    detection_ms: float
    memory_update_ms: float
    vlm_ms: Optional[float]
    rule_eval_ms: float
    tts_ms: Optional[float]
    total_ms: float
    fps: float
    ram_mb: float
```

## Module Interfaces

```python
# Abstract interfaces for each component

class BaseDetector:
    def detect(self, frame: np.ndarray) -> list[Detection]: ...
    def warmup(self) -> None: ...
    def get_class_names(self) -> dict[int, str]: ...

class BaseMemory:
    def update(self, detections: list[Detection]) -> None: ...
    def get_active(self, max_age_frames: int) -> list[MemoryEntry]: ...
    def is_present(self, class_id: int) -> bool: ...
    def get_duration(self, class_id: int) -> int: ...  # frames present

class BaseAnalyzer:
    def analyze(self, frame: np.ndarray, detections: list[Detection]) -> SceneContext: ...
    def is_available(self) -> bool: ...

class BaseRuleEngine:
    def evaluate(self, detections: list[Detection], memory: BaseMemory,
                 context: Optional[SceneContext]) -> list[Alert]: ...
    def reload_rules(self) -> None: ...

class BaseTTS:
    def speak(self, text: str) -> None: ...
    def is_speaking(self) -> bool: ...
    def health_check(self) -> bool: ...

class BasePlugin:
    def on_detection(self, detections: list[Detection]) -> None: ...
    def on_alert(self, alert: Alert) -> None: ...
    def on_frame(self, frame: np.ndarray) -> None: ...
```
